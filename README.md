# SensitiveWord
 * 敏感词过滤类
 * 采用DFA算法
 装载后数据结构如下：
   $words = [
       '小' => [
           '日' => [
               '本' => false,
           ],
       ],
       '日' => [
          '本' => [
              '鬼' => [
                   '子' => false,
               ],
               '人' => false,
           ],
       ],
   ];
